239  bool check_limit(Vectorq7x1 *pose_to_check)
240  {

253  {
254  (*pose_to_check)[i] = pose_limit[1][i];
255  ROS_WARN("joint_pose[%d] is below lower limit", i);
256  all_correct = false;
257  }
258  }
259  ROS_INFO("Baxter_right_arm: pose after adjust: [%f, %f, %f, %f, %f, %f, %f]",
260  (*pose_to_check)[0], (*pose_to_check)[1], (*pose_to_check)[2], (*pose_to_check)[3],
261  (*pose_to_check)[4], (*pose_to_check)[5], (*pose_to_check)[6]);
